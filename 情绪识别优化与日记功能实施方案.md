**2. src/services/baiduAuth.ts - 百度API认证**

```typescript
/**
 * 百度AI API认证服务
 */

import axios from 'axios';

// Access Token缓存
let cachedToken: string | null = null;
let tokenExpireTime: number = 0;

/**
 * 获取百度AI Access Token
 * 文档：https://ai.baidu.com/ai-doc/REFERENCE/Ck3dwjhhu
 */
export async function getBaiduAccessToken(): Promise<string> {
  // 如果token还在有效期内，直接返回
  if (cachedToken && Date.now() < tokenExpireTime) {
    return cachedToken;
  }

  const apiKey = import.meta.env.VITE_BAIDU_API_KEY;
  const secretKey = import.meta.env.VITE_BAIDU_SECRET_KEY;

  if (!apiKey || !secretKey) {
    throw new Error('百度API配置缺失，请检查.env文件');
  }

  try {
    const response = await axios.post(
      'https://aip.baidubce.com/oauth/2.0/token',
      null,
      {
        params: {
          grant_type: 'client_credentials',
          client_id: apiKey,
          client_secret: secretKey,
        },
      }
    );

    if (response.data.access_token) {
      cachedToken = response.data.access_token;
      // Token有效期30天，这里设置为25天后过期（保守策略）
      tokenExpireTime = Date.now() + 25 * 24 * 60 * 60 * 1000;
      return cachedToken;
    } else {
      throw new Error('获取Access Token失败');
    }
  } catch (error) {
    console.error('百度API认证失败:', error);
    throw new Error('百度API认证失败，请检查API Key和Secret Key');
  }
}
```

**3. src/services/emotionAnalysis.ts - 情绪分析服务**

```typescript
/**
 * 情绪分析服务
 * 使用百度情感倾向分析API
 */

import axios from 'axios';
import { getBaiduAccessToken } from './baiduAuth';
import type {
  EmotionAnalysisResult,
  BaiduSentimentResponse,
  EmotionType,
  EmotionIntensity,
} from '../types/emotion';

/**
 * 情绪关键词映射（用于辅助判断具体情绪类型）
 */
const EMOTION_KEYWORDS: Record<EmotionType, RegExp> = {
  angry: /生气|愤怒|气死|烦死|讨厌|恨|吵架|骂|烦|火大|崩溃|受不了/,
  sad: /难过|伤心|哭|委屈|失落|沮丧|心痛|绝望|无助|痛苦/,
  anxious: /焦虑|担心|害怕|紧张|压力|喘不过气|失眠|不安|恐惧/,
  frustrated: /挫败|失败|做不到|不行|没用|差劲|失望|放弃/,
  confused: /不懂|疑惑|迷茫|纠结|不知道|怎么办|为什么/,
  understood: /理解|明白|原来|懂了|释然|恍然/,
  grateful: /感谢|谢谢|感激|幸好|还好|庆幸/,
  hopeful: /希望|会好|试试|可以|愿意|想通|决定|加油/,
  happy: /开心|高兴|快乐|兴奋|好|棒|赞|哈哈|嘿嘿/,
  neutral: /.*/,
};

/**
 * 调用百度情感倾向分析API
 */
async function callBaiduSentimentAPI(text: string): Promise<BaiduSentimentResponse> {
  const accessToken = await getBaiduAccessToken();
  const url = `https://aip.baidubce.com/rpc/2.0/nlp/v1/sentiment_classify?access_token=${accessToken}`;

  try {
    const response = await axios.post(url, {
      text: text,
    });

    if (response.data.error_code) {
      throw new Error(`百度API错误: ${response.data.error_msg}`);
    }

    return response.data;
  } catch (error) {
    console.error('调用百度情感分析API失败:', error);
    throw error;
  }
}

/**
 * 根据文本内容和情感倾向判断具体情绪类型
 */
function detectEmotionType(text: string, sentiment: number): EmotionType {
  // 优先通过关键词匹配
  for (const [emotion, regex] of Object.entries(EMOTION_KEYWORDS)) {
    if (emotion !== 'neutral' && regex.test(text)) {
      return emotion as EmotionType;
    }
  }

  // 如果没有匹配到关键词，根据情感倾向返回基础情绪
  if (sentiment > 0.6) return 'happy';
  if (sentiment < -0.6) return 'sad';
  if (sentiment < -0.3) return 'frustrated';
  return 'neutral';
}

/**
 * 计算情绪强度等级
 */
function getIntensityLevel(intensity: number): EmotionIntensity {
  if (intensity >= 0.7) return 'high';
  if (intensity >= 0.4) return 'medium';
  return 'low';
}

/**
 * 主函数：分析文本情绪
 */
export async function analyzeEmotion(text: string): Promise<EmotionAnalysisResult> {
  // 如果文本过短，使用本地关键词匹配
  if (text.length < 5) {
    return {
      primaryEmotion: detectEmotionType(text, 0),
      intensity: 0.5,
      intensityLevel: 'medium',
      sentiment: 0,
      confidence: 0.6,
      text,
      timestamp: Date.now(),
    };
  }

  try {
    // 调用百度API
    const result = await callBaiduSentimentAPI(text);
    const item = result.items[0];

    // 计算情感倾向 (-1到1)
    const sentiment = item.positive_prob - item.negative_prob;

    // 情绪强度（取正负概率中的最大值）
    const intensity = Math.max(item.positive_prob, item.negative_prob);

    // 判断具体情绪类型
    const primaryEmotion = detectEmotionType(text, sentiment);

    return {
      primaryEmotion,
      intensity,
      intensityLevel: getIntensityLevel(intensity),
      sentiment,
      confidence: item.confidence,
      text,
      timestamp: Date.now(),
    };
  } catch (error) {
    console.error('情绪分析失败，使用降级方案:', error);

    // 降级方案：使用本地关键词匹配
    return {
      primaryEmotion: detectEmotionType(text, 0),
      intensity: 0.5,
      intensityLevel: 'medium',
      sentiment: 0,
      confidence: 0.5,
      text,
      timestamp: Date.now(),
    };
  }
}

/**
 * 批量分析情绪（用于情绪日记分析）
 */
export async function analyzeBatchEmotions(texts: string[]): Promise<EmotionAnalysisResult[]> {
  // 为了避免并发过多，分批处理
  const results: EmotionAnalysisResult[] = [];
  const batchSize = 5;

  for (let i = 0; i < texts.length; i += batchSize) {
    const batch = texts.slice(i, i + batchSize);
    const batchResults = await Promise.all(
      batch.map(text => analyzeEmotion(text))
    );
    results.push(...batchResults);

    // 避免API限流，每批间隔100ms
    if (i + batchSize < texts.length) {
      await new Promise(resolve => setTimeout(resolve, 100));
    }
  }

  return results;
}
```

```typescript
// src/utils/debounce.ts

export function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout;
  return function executedFunction(...args: Parameters<T>) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
}
```

```typescript
// src/db/index.ts

import { openDB, DBSchema, IDBPDatabase } from 'idb';
import type { EmotionJournalEntry, EmotionStats } from './schema';

interface SoulHugDB extends DBSchema {
  journals: {
    key: string;
    value: EmotionJournalEntry;
    indexes: {
      'by-date': string;
      'by-user': string;
      'by-emotion': string;
    };
  };
  stats: {
    key: string;
    value: EmotionStats;
    indexes: {
      'by-date': string;
    };
  };
}

let db: IDBPDatabase<SoulHugDB>;

export async function initDB() {
  db = await openDB<SoulHugDB>('soulhug-db', 1, {
    upgrade(db) {
      // 创建journals表
      const journalStore = db.createObjectStore('journals', {
        keyPath: 'id',
      });
      journalStore.createIndex('by-date', 'date');
      journalStore.createIndex('by-user', 'userId');
      journalStore.createIndex('by-emotion', 'emotion.primaryEmotion');

      // 创建stats表
      const statsStore = db.createObjectStore('stats', {
        keyPath: ['userId', 'date'],
      });
      statsStore.createIndex('by-date', 'date');
    },
  });

  return db;
}

export function getDB() {
  if (!db) {
    throw new Error('Database not initialized. Call initDB() first.');
  }
  return db;
}
```

```typescript
  await db.add('journals', journal);

  // 更新统计数据
  await updateStats(journal);

  return journal;
}

/**
 * 获取日记（按日期）
 */
export async function getJournalsByDate(date: string): Promise<EmotionJournalEntry[]> {
  const db = getDB();
  return await db.getAllFromIndex('journals', 'by-date', date);
}

 * 获取日记（按时间范围）
 */
export async function getJournalsByDateRange(
  startDate: string,
  endDate: string
): Promise<EmotionJournalEntry[]> {
  const db = getDB();
  const allJournals = await db.getAll('journals');

  return allJournals.filter(
    journal => journal.date >= startDate && journal.date <= endDate
  );
}

 * 更新日记
 */
export async function updateJournal(
  id: string,
  updates: Partial<EmotionJournalEntry>
): Promise<void> {

  const journal = await db.get('journals', id);

  if (!journal) {
    throw new Error(`Journal ${id} not found`);
  }

  const updated = {
    ...journal,
    ...updates,
    updatedAt: Date.now(),
  };

  await db.put('journals', updated);
}

 * 删除日记
 */

/**
 * 更新统计数据
 */
async function updateStats(journal: EmotionJournalEntry): Promise<void> {
  const db = getDB();
  const key = `${journal.userId}-${journal.date}`;

  let stats = await db.get('stats', key);

  if (!stats) {
    stats = {
      userId: journal.userId,
      date: journal.date,
      emotions: {} as Record<string, number>,
      totalEntries: 0,
      averageMood: 0,
    };
  }

  // 更新情绪统计
  const emotion = journal.emotion.primaryEmotion;
  stats.emotions[emotion] = (stats.emotions[emotion] || 0) + 1;

  // 更新总数
  stats.totalEntries += 1;

  // 重新计算平均心情（需要查询当天所有日记）
  const todayJournals = await getJournalsByDate(journal.date);
  const totalMood = todayJournals.reduce((sum, j) => sum + j.mood, 0);
  stats.averageMood = totalMood / todayJournals.length;

  await db.put('stats', stats);
}
```

```typescript
  dominantEmotion: EmotionType;
  emotionCounts: Record<EmotionType, number>;
  insights: string[];
  topTags: Array<{ tag: string; count: number }>;
}

export function WeeklyReport() {
  const [reportData, setReportData] = useState<WeeklyReportData | null>(null);
  const [isGenerating, setIsGenerating] = useState(false);

  useEffect(() => {
    generateReport();
  }, []);

  const generateReport = async () => {
    setIsGenerating(true);

    try {
      // 获取最近7天的日记
      const endDate = new Date();
      const startDate = new Date();
      startDate.setDate(endDate.getDate() - 7);

      const journals = await getJournalsByDateRange(
        startDate.toISOString().split('T')[0],
        endDate.toISOString().split('T')[0]
      );

      if (journals.length === 0) {
        setReportData(null);
        return;
      }

      // 统计数据
      const totalJournals = journals.length;
      const totalMood = journals.reduce((sum, j) => sum + j.mood, 0);
      const averageMood = totalMood / totalJournals;

      // 统计情绪
      const emotionCounts: Record<EmotionType, number> = {} as any;
      journals.forEach(j => {
        const emotion = j.emotion.primaryEmotion;
        emotionCounts[emotion] = (emotionCounts[emotion] || 0) + 1;
      });

      const dominantEmotion = Object.entries(emotionCounts).sort(
        ([, a], [, b]) => b - a
      )[0][0] as EmotionType;

      // 统计标签
      const tagCounts: Record<string, number> = {};
      journals.forEach(j => {
        j.tags.forEach(tag => {
          tagCounts[tag] = (tagCounts[tag] || 0) + 1;
        });
      });

      const topTags = Object.entries(tagCounts)
        .sort(([, a], [, b]) => b - a)
        .slice(0, 5)
        .map(([tag, count]) => ({ tag, count }));

      // 生成洞察
      const insights = generateInsights(journals, averageMood, dominantEmotion);

      setReportData({
        totalJournals,
        averageMood,
        dominantEmotion,
        emotionCounts,
        insights,
        topTags,
      });
    } catch (error) {
      console.error('生成报告失败:', error);
    } finally {
      setIsGenerating(false);
    }
  };

  const generateInsights = (
    journals: EmotionJournalEntry[],
    avgMood: number,
    dominantEmotion: EmotionType
  ): string[] => {
    const insights: string[] = [];

    // 洞察1：整体心情
    if (avgMood >= 4) {
      insights.push('🌟 本周你的整体心情很不错，继续保持！');
    } else if (avgMood >= 3) {
      insights.push('😊 本周你的心情比较平稳');
    } else {
      insights.push('💙 本周情绪有些低落，需要好好照顾自己');
    }

    // 洞察2：主要情绪
    const emotionLabel = EMOTION_LABELS[dominantEmotion];
    insights.push(`本周你最常感到${emotionLabel}，这占了你情绪的主要部分`);

    // 洞察3：情绪波动
    const moods = journals.map(j => j.mood);
    const maxMood = Math.max(...moods);
    const minMood = Math.min(...moods);
    if (maxMood - minMood >= 3) {
      insights.push('⚡ 本周情绪波动较大，建议多做放松练习');
    }

    // 洞察4：记录频率
    if (journals.length >= 5) {
      insights.push('✍️ 你坚持记录了' + journals.length + '天日记，很棒的习惯！');
    } else {
      insights.push('📝 试着每天都记录一下，会更好地了解自己哦');
    }

    return insights;
  };

  if (isGenerating) {
    return <div className="text-center py-8">生成报告中...</div>;
  }

  if (!reportData) {
    return (
      <div className="text-center py-8 text-gray-500">
        本周还没有日记记录，快去写一篇吧！
      </div>
    );
  }

  const emotionDisplay = getEmotionDisplay(reportData.dominantEmotion);

  return (
    <div className="max-w-2xl mx-auto p-6 bg-white rounded-lg shadow-lg">
      <h2 className="text-2xl font-bold mb-6">📊 本周情绪报告</h2>

      {/* 关键数据 */}
      <div className="grid grid-cols-3 gap-4 mb-6">
        <div className="text-center p-4 bg-purple-50 rounded-lg">
          <div className="text-3xl font-bold text-purple-600">
            {reportData.totalJournals}
          </div>
          <div className="text-sm text-gray-600 mt-1">日记篇数</div>
        </div>
        <div className="text-center p-4 bg-blue-50 rounded-lg">
          <div className="text-3xl font-bold text-blue-600">
            {reportData.averageMood.toFixed(1)}
          </div>
          <div className="text-sm text-gray-600 mt-1">平均心情</div>
        </div>
        <div className="text-center p-4 bg-pink-50 rounded-lg">
          <div className="text-3xl">{emotionDisplay.emoji}</div>
          <div className="text-sm text-gray-600 mt-1">主要情绪</div>
        </div>
      </div>

      {/* 情绪分布 */}
      <div className="mb-6">
        <h3 className="font-semibold mb-3">情绪分布</h3>
        {Object.entries(reportData.emotionCounts).map(([emotion, count]) => {
          const display = getEmotionDisplay(emotion as EmotionType);
          const percentage = (count / reportData.totalJournals) * 100;

          return (
            <div key={emotion} className="mb-2">
              <div className="flex justify-between text-sm mb-1">
                <span>{display.emoji} {display.label}</span>
                <span>{count}次 ({percentage.toFixed(0)}%)</span>
              </div>
              <div className="h-2 bg-gray-200 rounded-full overflow-hidden">
                <div
                  className="h-full rounded-full"
                  style={{
                    width: `${percentage}%`,
                    backgroundColor: display.color,
                  }}
                />
              </div>
            </div>
          );
        })}
      </div>

      {/* AI洞察 */}
      <div className="mb-6">
        <h3 className="font-semibold mb-3">💡 洞察与建议</h3>
        <ul className="space-y-2">
          {reportData.insights.map((insight, index) => (
            <li key={index} className="flex items-start gap-2">
              <span className="text-purple-500 mt-1">•</span>
              <span className="flex-1">{insight}</span>
            </li>
          ))}
        </ul>
      </div>

      {/* 热门标签 */}
      {reportData.topTags.length > 0 && (
        <div>
          <h3 className="font-semibold mb-3">🏷️ 本周热门话题</h3>
          <div className="flex flex-wrap gap-2">
            {reportData.topTags.map(({ tag, count }) => (
              <span
                key={tag}
                className="px-3 py-1 bg-purple-100 text-purple-700 rounded-full text-sm"
              >
                {tag} ({count})
              </span>
            ))}
          </div>
        </div>
      )}

      {/* 分享按钮 */}
      <div className="mt-6 flex gap-3">
        <button className="flex-1 py-2 bg-purple-500 text-white rounded-lg hover:bg-purple-600">
          导出报告
        </button>
        <button className="flex-1 py-2 bg-gray-200 text-gray-700 rounded-lg hover:bg-gray-300">
          分享给家长
        </button>
      </div>
    </div>
  );
}
```
